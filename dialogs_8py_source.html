45 # lk: because you will eventually find yourself back here.
46 # on QDialogs the Esc key closes the dialog through keyPressEvent(),
47 # which ultimately call `reject()`. Pressing the Enter/Return button
48 # is a shortcut for pressing the default button and thus calling `accept()`
49 # In turn both `accept()` and `reject()` evetually call `done()`.
50
51 # In the base dialog ignore both. In the subclasses, call the method
52 # from QDialog if required, not this one.
53 # `super(BaseDialog, self).keyPressEvent(a0)`
54 def keyPressEvent(self, a0: QKeyEvent) -> None:
55 if a0.matches(QKeySequence.Cancel):
56 a0.ignore()
57 return
58 if a0.key() == Qt.Key_Enter or a0.key() == Qt.Key_Return:
59 a0.ignore()
60 return
61 super().keyPressEvent(a0)
