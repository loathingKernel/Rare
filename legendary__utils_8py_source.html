102 def run(self):
103 if not self.corecore.is_installed(self.app_nameapp_name):
104 logger.error(f'Game "{self.app_name}" is not installed')

106
107 self.signalssignals.started.emit(self.app_nameapp_name)
108 logger.info(f'Loading installed manifest for "{self.app_name}"')
109 igame = self.corecore.get_installed_game(self.app_nameapp_name)
110 manifest_data, _ = self.corecore.get_installed_manifest(self.app_nameapp_name)
111 manifest = self.corecore.load_manifest(manifest_data)
112
113 files = sorted(manifest.file_manifest_list.elements, key=lambda a: a.filename.lower())
114
115 # build list of hashes
116 file_list = [(f.filename, f.sha_hash.hex()) for f in files]
117 total = len(file_list)
118 num = 0
119 failed = []
120 missing = []
121
122 logger.info(f'Verifying "{igame.title}" version "{manifest.meta.build_version}"')
123 repair_file = []
124 for result, path, result_hash, _ in validate_files(igame.install_path, file_list):
125 num += 1
126 self.signalssignals.status.emit(num, total, self.app_nameapp_name)
127
128 if result == VerifyResult.HASH_MATCH:
129 repair_file.append(f"{result_hash}:{path}")
130 continue
131 elif result == VerifyResult.HASH_MISMATCH:
132 logger.error(f'File does not match hash: "{path}"')
133 repair_file.append(f"{result_hash}:{path}")
134 failed.append(path)
135 elif result == VerifyResult.FILE_MISSING:
136 logger.error(f'File is missing: "{path}"')
137 missing.append(path)
138 else:
139 logger.error(f'Other failure (see log), treating file as missing: "{path}"')
140 missing.append(path)
141
142 # always write repair file, even if all match
143 if repair_file:
144 repair_filename = os.path.join(self.corecore.lgd.get_tmp_path(), f"{self.app_name}.repair")
145 with open(repair_filename, "w") as f:
146 f.write("\n".join(repair_file))
147 logger.debug(f'Written repair file to "{repair_filename}"')
148
149 if not missing and not failed:
150 logger.info("Verification finished successfully.")
151 else:
152 logger.warning(
153 f"Verification failed, {len(failed)} file(s) corrupted, {len(missing)} file(s) are missing."
154 )
155 self.signalssignals.summary.emit(len(failed), len(missing), self.app_nameapp_name)
