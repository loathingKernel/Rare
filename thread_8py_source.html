160 def _handle_postinstall(self, postinstall, igame):
161 logger.info("This game lists the following prerequisites to be installed:")
162 logger.info(f'- {postinstall["name"]}: {" ".join((postinstall["path"], postinstall["args"]))}')
163 if platform.system() == "Windows":
164 if self.item.options.install_prereqs:
165 logger.info("Launching prerequisite executable..")
166 self.core.prereq_installed(igame.app_name)
167 req_path, req_exec = os.path.split(postinstall["path"])
168 work_dir = os.path.join(igame.install_path, req_path)
169 fullpath = os.path.join(work_dir, req_exec)
170 proc = QProcess()
171 proc.setProcessChannelMode(QProcess.MergedChannels)
172 proc.readyReadStandardOutput.connect(
173 lambda: logger.debug(str(proc.readAllStandardOutput().data(), "utf-8", "ignore"))
174 )
175 proc.setProgram(fullpath)
176 proc.setArguments(postinstall.get("args", "").split(" "))
177 proc.setWorkingDirectory(work_dir)
178 proc.start()
179 proc.waitForFinished() # wait, because it is inside the thread
180 else:
181 logger.info("Automatic installation not available on Linux.")
