83 logger.error("Got unexpected action")
84 elif action == message_models.Actions.finished:

89 elif action == message_models.Actions.error:

93 "Error in game {}: \n{}").format(self.gamegame.app_title, model.error_string))
94 elif action == message_models.Actions.state_update:
