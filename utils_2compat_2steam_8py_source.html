33# Notes:
34# Anything older than 'Proton 5.13' doesn't have the 'require_tool_appid' attribute.
35# Anything older than 'Proton 7.0' doesn't have the 'compatmanager_layer_name' attribute.
36# In addition to that, the 'Steam Linux Runtime 1.0 (scout)' runtime lists the
37# 'Steam Linux Runtime 2.0 (soldier)' runtime as a dependency and is probably what was
38# being used for any version before 5.13.
39#
40# As a result the following implementation will list versions from 7.0 onwards which honestly
41# is a good trade-off for the amount of complexity supporting everything would ensue.

74 def command(self, verb: SteamVerb = SteamVerb.DEFAULT) -> List[str]:
75 tool_path = os.path.normpath(self.tool_pathtool_path)
76 cmd = "".join([shlex.quote(tool_path), self.toolmanifest["commandline"]])
77 # NOTE: "waitforexitandrun" seems to be the verb used in by steam to execute stuff
78 # `run` is used when setting up the environment, so use that if we are setting up the prefix.
79 cmd = cmd.replace("%verb%", str(verb))
80 return shlex.split(cmd)

82 def as_str(self, verb: SteamVerb = SteamVerb.DEFAULT):
83 return " ".join(map(shlex.quote, self.commandcommand(verb)))

149def find_runtimes(steam_path: str, library: str) -> Dict[str, SteamRuntime]:
150 runtimes = {}
151 appmanifests = find_appmanifests(library)
152 common = os.path.join(library, "common")
153 for appmanifest in appmanifests:
154 folder = appmanifest["AppState"]["installdir"]
155 tool_path = os.path.join(common, folder)
156 if os.path.isfile(vdf_file := os.path.join(tool_path, "toolmanifest.vdf")):
157 with open(vdf_file, "r") as f:
158 toolmanifest = vdf.load(f)
159 if toolmanifest["manifest"]["compatmanager_layer_name"] == "container-runtime":
160 runtimes.update(
161 {
162 appmanifest["AppState"]["appid"]: SteamRuntime(
163 steam_path=steam_path,
164 steam_library=library,
165 appmanifest=appmanifest,
166 tool_path=tool_path,
167 toolmanifest=toolmanifest["manifest"],
168 )
169 }
170 )

197def find_compatibility_tools(steam_path: str) -> List[CompatibilityTool]:
198 compatibilitytools_paths = {
199 "/usr/share/steam/compatibilitytools.d",
200 os.path.expanduser(os.path.join(steam_path, "compatibilitytools.d")),
201 os.path.expanduser("~/.steam/compatibilitytools.d"),
202 os.path.expanduser("~/.steam/root/compatibilitytools.d"),
203 }
204 compatibilitytools_paths = {
205 os.path.realpath(path) for path in compatibilitytools_paths if os.path.isdir(path)
206 }
207 tools = []
208 for path in compatibilitytools_paths:
209 for entry in os.scandir(path):
210 entry_path = os.path.join(path, entry.name)
211 if entry.is_dir():
212 tool_vdf = os.path.join(entry_path, "compatibilitytool.vdf")
213 elif entry.is_file() and entry.name.endswith(".vdf"):
214 tool_vdf = entry_path
215 else:
216 continue
217
218 if not os.path.isfile(tool_vdf):
219 continue
220
221 with open(tool_vdf, "r") as f:
222 compatibilitytool = vdf.load(f)
223
224 entry_tools = compatibilitytool["compatibilitytools"]["compat_tools"]
225 for entry_tool in entry_tools.values():
226 if entry_tool["from_oslist"] != "windows" and entry_tool["to_oslist"] != "linux":
227 continue
228
229 install_path = entry_tool["install_path"]
230 tool_path = os.path.abspath(os.path.join(os.path.dirname(tool_vdf), install_path))
231 manifest_vdf = os.path.join(tool_path, "toolmanifest.vdf")
232
233 if not os.path.isfile(manifest_vdf):
234 continue
235
236 with open(manifest_vdf, "r") as f:
237 manifest = vdf.load(f)
238
239 tools.append(
240 CompatibilityTool(
241 steam_path=steam_path,
242 tool_path=tool_path,
243 toolmanifest=manifest["manifest"],
244 compatibilitytool=entry_tool,
245 )
246 )
247 return tools

277def get_steam_environment(
278 tool: Optional[Union[ProtonTool, CompatibilityTool]] = None, compat_path: Optional[str] = None
279) -> Dict:
280 # If the tool is unset, return all affected env variable names
281 # IMPORTANT: keep this in sync with the code below
282 environ = {"STEAM_COMPAT_DATA_PATH": compat_path if compat_path else ""}
283 if tool is None:
284 environ["STEAM_COMPAT_DATA_PATH"] = ""
285 environ["STEAM_COMPAT_CLIENT_INSTALL_PATH"] = ""
286 environ["STEAM_COMPAT_LIBRARY_PATHS"] = ""
287 environ["STEAM_COMPAT_MOUNTS"] = ""
288 environ["STEAM_COMPAT_TOOL_PATHS"] = ""
289 return environ
290
291 environ["STEAM_COMPAT_CLIENT_INSTALL_PATH"] = tool.steam_path
292 if isinstance(tool, ProtonTool):
293 environ["STEAM_COMPAT_LIBRARY_PATHS"] = tool.steam_library
294 if tool.runtime is not None:
295 compat_mounts = [tool.tool_path, tool.runtime.tool_path]
296 environ["STEAM_COMPAT_MOUNTS"] = ":".join(compat_mounts)
297 tool_paths = [tool.tool_path]
298 if tool.runtime is not None:
299 tool_paths.append(tool.runtime.tool_path)
300 environ["STEAM_COMPAT_TOOL_PATHS"] = ":".join(tool_paths)
301 return environ

304def _find_tools() -> List[Union[ProtonTool, CompatibilityTool]]:
305 steam_path = find_steam()
306 if steam_path is None:
307 logger.info("Steam could not be found")
308 return []
309 logger.info("Found Steam in %s", steam_path)
310
311 steam_libraries = find_libraries(steam_path)
312 logger.debug("Searching for tools in libraries:")
313 logger.debug("%s", steam_libraries)
314
315 runtimes = {}
316 for library in steam_libraries:
317 runtimes.update(find_runtimes(steam_path, library))
318
319 tools = []
320 for library in steam_libraries:
321 tools.extend(find_protons(steam_path, library))
322 tools.extend(find_compatibility_tools(steam_path))
323
324 for tool in tools:
325 runtime = get_runtime(tool, runtimes)
326 tool.runtime = runtime
327
328 tools = list(filter(lambda t: bool(t), tools))
329
330 return tools
