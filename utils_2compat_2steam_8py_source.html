34# Notes:
35# Anything older than 'Proton 5.13' doesn't have the 'require_tool_appid' attribute.
36# Anything older than 'Proton 7.0' doesn't have the 'compatmanager_layer_name' attribute.
37# In addition to that, the 'Steam Linux Runtime 1.0 (scout)' runtime lists the
38# 'Steam Linux Runtime 2.0 (soldier)' runtime as a dependency and is probably what was
39# being used for any version before 5.13.
40#
41# As a result the following implementation will list versions from 7.0 onwards which honestly
42# is a good trade-off for the amount of complexity supporting everything would ensue.

75 def command(self, verb: SteamVerb = SteamVerb.DEFAULT) -> List[str]:
76 tool_path = os.path.normpath(self.tool_pathtool_path)
77 cmd = "".join([shlex.quote(tool_path), self.toolmanifest["commandline"]])
78 # NOTE: "waitforexitandrun" seems to be the verb used in by steam to execute stuff
79 # `run` is used when setting up the environment, so use that if we are setting up the prefix.
80 cmd = cmd.replace("%verb%", verb.value)
81 return shlex.split(cmd)

249def find_compatibility_tools(steam_path: str) -> List[CompatibilityTool]:
250 compatibilitytools_paths = {
251 "/usr/share/steam/compatibilitytools.d",
252 os.path.expanduser(os.path.join(steam_path, "compatibilitytools.d")),
253 os.path.expanduser("~/.steam/compatibilitytools.d"),
254 os.path.expanduser("~/.steam/root/compatibilitytools.d"),
255 }
256 compatibilitytools_paths = {
257 os.path.realpath(path) for path in compatibilitytools_paths if os.path.isdir(path)
258 }
259 tools = []
260 for path in compatibilitytools_paths:
261 for entry in os.scandir(path):
262 entry_path = os.path.join(path, entry.name)
263 if entry.is_dir():
264 tool_vdf = os.path.join(entry_path, "compatibilitytool.vdf")
265 elif entry.is_file() and entry.name.endswith(".vdf"):
266 tool_vdf = entry_path
267 else:
268 continue
269
270 if not os.path.isfile(tool_vdf):
271 continue
272
273 with open(tool_vdf, "r", encoding="utf-8") as f:
274 compatibilitytool = vdf.load(f)
275
276 entry_tools = compatibilitytool["compatibilitytools"]["compat_tools"]
277 for entry_tool in entry_tools.values():
278 if entry_tool["from_oslist"] != "windows" and entry_tool["to_oslist"] != "linux":
279 continue
280
281 install_path = entry_tool["install_path"]
282 tool_path = os.path.abspath(os.path.join(os.path.dirname(tool_vdf), install_path))
283 manifest_vdf = os.path.join(tool_path, "toolmanifest.vdf")
284
285 if not os.path.isfile(manifest_vdf):
286 continue
287
288 with open(manifest_vdf, "r", encoding="utf-8") as f:
289 manifest = vdf.load(f)
290
291 tools.append(
292 CompatibilityTool(
293 steam_path=steam_path,
294 tool_path=tool_path,
295 toolmanifest=manifest["manifest"],
296 compatibilitytool=entry_tool,
297 )
298 )
299 return tools

329def get_steam_environment(
330 tool: Optional[Union[ProtonTool, CompatibilityTool]] = None, compat_path: Optional[str] = None
331) -> Dict:
332 # If the tool is unset, return all affected env variable names
333 # IMPORTANT: keep this in sync with the code below
334 environ = {"STEAM_COMPAT_DATA_PATH": compat_path if compat_path else ""}
335 if tool is None:
336 environ["STEAM_COMPAT_DATA_PATH"] = ""
337 environ["STEAM_COMPAT_CLIENT_INSTALL_PATH"] = ""
338 environ["STEAM_COMPAT_LIBRARY_PATHS"] = ""
339 environ["STEAM_COMPAT_MOUNTS"] = ""
340 environ["STEAM_COMPAT_TOOL_PATHS"] = ""
341 environ["PROTON_EAC_RUNTIME"] = ""
342 environ["PROTON_BATTLEYE_RUNTIME"] = ""
343 return environ
344
345 environ["STEAM_COMPAT_CLIENT_INSTALL_PATH"] = tool.steam_path
346 if isinstance(tool, ProtonTool):
347 environ["STEAM_COMPAT_LIBRARY_PATHS"] = tool.steam_library
348 if tool.runtime is not None:
349 compat_mounts = [tool.tool_path, tool.runtime.tool_path]
350 environ["STEAM_COMPAT_MOUNTS"] = ":".join(compat_mounts)
351 tool_paths = [tool.tool_path]
352 if tool.runtime is not None:
353 tool_paths.append(tool.runtime.tool_path)
354 environ["STEAM_COMPAT_TOOL_PATHS"] = ":".join(tool_paths)
355 if tool.anticheat is not None:
356 if (appid := ANTICHEAT_RUNTIMES["eac_runtime"]) in tool.anticheat.keys():
357 environ["PROTON_EAC_RUNTIME"] = tool.anticheat[appid].tool_path
358 if (appid := ANTICHEAT_RUNTIMES["battleye_runtime"]) in tool.anticheat.keys():
359 environ["PROTON_BATTLEYE_RUNTIME"] = tool.anticheat[appid].tool_path
360 return environ

363def _find_tools() -> List[Union[ProtonTool, CompatibilityTool]]:
364 steam_path = find_steam()
365 if steam_path is None:
366 logger.info("Steam could not be found")
367 return []
368 logger.info("Found Steam in %s", steam_path)
369
370 steam_libraries = find_libraries(steam_path)
371 logger.debug("Searching for tools in libraries:")
372 logger.debug("%s", steam_libraries)
373
374 runtimes = {}
375 for library in steam_libraries:
376 runtimes.update(find_runtimes(steam_path, library))
377
378 anticheat = {}
379 for library in steam_libraries:
380 anticheat.update(find_anticheat(steam_path, library))
381
382 tools = []
383 for library in steam_libraries:
384 tools.extend(find_protons(steam_path, library))
385 tools.extend(find_compatibility_tools(steam_path))
386
387 for tool in tools:
388 runtime = get_runtime(tool, runtimes)
389 tool.runtime = runtime
390 if tool.layer == "proton":
391 tool.anticheat = anticheat
392
393 tools = list(filter(lambda t: bool(t), tools))
394
395 return tools
