36def find_app_name(path: str, core) -> Optional[str]:
37 if os.path.exists(os.path.join(path, ".egstore")):
38 for i in os.listdir(os.path.join(path, ".egstore")):
39 if i.endswith(".mancpn"):
40 with open(os.path.join(path, ".egstore", i)) as file:
41 app_name = json.load(file).get("AppName")
42 return app_name
43 elif app_name := LegendaryCLI(core)._resolve_aliases(os.path.basename(os.path.normpath(path))):
44 # return None if game does not exist (Workaround for overlay)
45 if not core.get_game(app_name):
46 return None
47 return app_name
48 else:
49 logger.warning(f"Could not find AppName for {path}")
50 return None

98 folders = [i for i in self.pathpath.iterdir() if i.is_dir()]
99 number_of_folders = len(folders)

112 def __try_import(self, path: Path, app_name: str = None) -> ImportedGame:
113 result = ImportedGame(ImportResult.ERROR)
114 result.path = str(path)
115 if app_name or (app_name := find_app_name(str(path), self.corecore)):
116 game = self.corecore.get_game(app_name)
117 result.app_name = app_name
118 result.app_title = game.app_title
119 platform = self.platformplatform
120 if platform not in self.corecore.get_game(app_name, update_meta=False).asset_infos:
121 platform = "Windows"
122 success, message = self.__import_game__import_game(path, app_name, platform)
123 if not success:
124 result.result = ImportResult.FAILED
125 result.message = message
126 else:
127 result.result = ImportResult.SUCCESS
128 return result
