38def find_app_name(path: str, core) -> Optional[str]:
39 if os.path.exists(os.path.join(path, ".egstore")):
40 for i in os.listdir(os.path.join(path, ".egstore")):
41 if i.endswith(".mancpn"):
42 with open(os.path.join(path, ".egstore", i)) as file:
43 app_name = json.load(file).get("AppName")
44 return app_name
45 elif app_name := LegendaryCLI(core)._resolve_aliases(os.path.basename(os.path.normpath(path))):
46 # return None if game does not exist (Workaround for overlay)
47 if not core.get_game(app_name):
48 return None
49 return app_name
50 else:
51 logger.warning(f"Could not find AppName for {path}")
52 return None

114 def __try_import(self, path: Path, app_name: str = None) -> ImportedGame:
115 result = ImportedGame(ImportResult.ERROR)
116 result.path = str(path)
117 if app_name or (app_name := find_app_name(str(path), self.corecore)):
118 game = self.corecore.get_game(app_name)
119 result.app_name = app_name
120 result.app_title = game.app_title
121 platform = self.platformplatform
122 if platform not in self.corecore.get_game(app_name, update_meta=False).asset_infos:
123 platform = "Windows"
124 success, message = self.__import_game__import_game(path, app_name, platform)
125 if not success:
126 result.result = ImportResult.FAILED
127 result.message = message
128 else:
129 result.result = ImportResult.SUCCESS
130 return result
