334 image = QImage()
335 image.loadFromData(image_data)
336 image.convertToFormat(QImage.Format.Format_ARGB32_Premultiplied)
337 # lk: Images are not always at the correct aspect ratio, so crop them to size
338 wr, hr = preset.aspect_ratio
339 factor = min(image.width() // wr, image.height() // hr)
340 rem_w = (image.width() - factor * wr) // 2
341 rem_h = (image.height() - factor * hr) // 2
342 image = image.copy(rem_w, rem_h, factor * wr, factor * hr)
343
344 if logo_data is not None:
345 logo = QImage()
346 logo.loadFromData(logo_data)
347 logo.convertToFormat(QImage.Format.Format_ARGB32_Premultiplied)
348 if logo.width() > image.width():
349 logo = logo.scaled(
350 image.width(),
351 image.height(),
352 Qt.AspectRatioMode.KeepAspectRatio,
353 Qt.TransformationMode.SmoothTransformation,
354 )
355 painter = QPainter(image)
356 painter.drawImage(
357 (image.width() - logo.width()) // 2,
358 image.height() - logo.height(),
359 logo,

361 painter.end()
