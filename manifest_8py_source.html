14def read_fstring(bio):
15 length = struct.unpack('<i', bio.read(4))[0]
16
17 # if the length is negative the string is UTF-16 encoded, this was a pain to figure out.
18 if length < 0:
19 # utf-16 chars are (generally) 2 bytes wide, but the length is # of characters, not bytes.
20 # 4-byte wide chars exist, but best I can tell Epic's (de)serializer doesn't support those.
21 length *= -2
22 s = bio.read(length - 2).decode('utf-16')
23 bio.seek(2, 1) # utf-16 strings have two byte null terminators
24 elif length > 0:
25 s = bio.read(length - 1).decode('ascii')
26 bio.seek(1, 1) # skip string null terminator
27 else: # empty string, no terminators or anything
28 s = ''
29
30 return s

33def write_fstring(bio, string):
34 if not string:
35 bio.write(struct.pack('<i', 0))
36 return
37
38 try:
39 s = string.encode('ascii')
40 bio.write(struct.pack('<i', len(string) + 1))
41 bio.write(s)
42 bio.write(b'\x00')
43 except UnicodeEncodeError:
44 s = string.encode('utf-16le')
45 bio.write(struct.pack('<i', -(len(string) + 1)))
46 bio.write(s)
47 bio.write(b'\x00\x00')

490 @classmethod
491 def read(cls, bio):
492 fml_start = bio.tell()
493 _fml = cls()
494 _fml.size = struct.unpack('<I', bio.read(4))[0]
495 _fml.version = struct.unpack('B', bio.read(1))[0]
496 _fml.count = struct.unpack('<I', bio.read(4))[0]
497
498 for i in range(_fml.count):
499 _fml.elements.append(FileManifest())
500
501 for fm in _fml.elements:
502 fm.filename = read_fstring(bio)
503
504 # never seen this used in any of the manifests I checked but can't wait for something to break because of it
505 for fm in _fml.elements:
506 fm.symlink_target = read_fstring(bio)
507
508 # For files this is actually the SHA1 instead of whatever it is for chunks...
509 for fm in _fml.elements:
510 fm.hash = bio.read(20)
511
512 # Flags, the only one I've seen is for executables
513 for fm in _fml.elements:
514 fm.flags = struct.unpack('B', bio.read(1))[0]
515
516 # install tags, no idea what they do, I've only seen them in the Fortnite manifest
517 for fm in _fml.elements:
518 _elem = struct.unpack('<I', bio.read(4))[0]
519 for i in range(_elem):
520 fm.install_tags.append(read_fstring(bio))
521
522 # Each file is made up of "Chunk Parts" that can be spread across the "chunk stream"
523 for fm in _fml.elements:
524 _elem = struct.unpack('<I', bio.read(4))[0]
525 _offset = 0
526 for i in range(_elem):
527 chunkp = ChunkPart()
528 _start = bio.tell()
529 _size = struct.unpack('<I', bio.read(4))[0]
530 chunkp.guid = struct.unpack('<IIII', bio.read(16))
531 chunkp.offset = struct.unpack('<I', bio.read(4))[0]
532 chunkp.size = struct.unpack('<I', bio.read(4))[0]
533 chunkp.file_offset = _offset
534 fm.chunk_parts.append(chunkp)
535 _offset += chunkp.size
536 if (diff := (bio.tell() - _start - _size)) > 0:
537 logger.warning(f'Did not read {diff} bytes from chunk part!')
538 bio.seek(diff)
539
540 # MD5 hash + MIME type
541 if _fml.version >= 1:
542 for fm in _fml.elements:
543 _has_md5 = struct.unpack('<I', bio.read(4))[0]
544 if _has_md5 != 0:
545 fm.hash_md5 = bio.read(16)
546
547 for fm in _fml.elements:
548 fm.mime_type = read_fstring(bio)
549
550 # SHA256 hash
551 if _fml.version >= 2:
552 for fm in _fml.elements:
553 fm.hash_sha256 = bio.read(32)
554
555 # we have to calculate the actual file size ourselves
556 for fm in _fml.elements:
557 fm.file_size = sum(c.size for c in fm.chunk_parts)
558
559 if (size_read := bio.tell() - fml_start) != _fml.size:
560 logger.warning(f'Did not read entire file data list! Version: {_fml.version}, '
561 f'{_fml.size - size_read} bytes missing, skipping...')
562 bio.seek(_fml.size - size_read, 1)
563 # downgrade version to prevent issues during serialisation
564 _fml.version = 0

573

576 for fm in self.elementselements:

580 for fm in self.elementselements:
