107 """
108 Handles getting an exchange code from a session id

116 'X-Epic-Strategy-Flags': '',

118 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) '

901 def check_savegame_state(self, path: str, save: SaveGameFile) -> (SaveGameStatus, (datetime, datetime)):
902 latest = 0
903 for _dir, _, _files in os.walk(path):
904 for _file in _files:
905 s = os.stat(os.path.join(_dir, _file))
906 latest = max(latest, s.st_mtime)
907
908 if not latest and not save:
909 return SaveGameStatus.NO_SAVE, (None, None)
910
911 # timezones are fun!
912 dt_local = datetime.fromtimestamp(latest).replace(tzinfo=self.local_timezonelocal_timezone).astimezone(timezone.utc)
913 if not save:
914 return SaveGameStatus.LOCAL_NEWER, (dt_local, None)
915
916 dt_remote = datetime.strptime(save.manifest_name, '%Y.%m.%d-%H.%M.%S.manifest').replace(tzinfo=timezone.utc)
917 if not latest:
918 return SaveGameStatus.REMOTE_NEWER, (None, dt_remote)
919
920 self.loglog.debug(f'Local save date: {str(dt_local)}, Remote save date: {str(dt_remote)}')
921
922 # Ideally we check the files themselves based on manifest,
923 # this is mostly a guess but should be accurate enough.
924 if abs((dt_local - dt_remote).total_seconds()) < 60:
925 return SaveGameStatus.SAME_AGE, (dt_local, dt_remote)
926 elif dt_local > dt_remote:
927 return SaveGameStatus.LOCAL_NEWER, (dt_local, dt_remote)
928 else:
929 return SaveGameStatus.REMOTE_NEWER, (dt_local, dt_remote)
