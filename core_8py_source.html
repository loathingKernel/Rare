32 def get_cdn_manifest(self, game, platform='Windows', disable_https=False):
33 manifest_urls, base_urls, manifest_hash = self.get_cdn_urls(game, platform)
34
35 if disable_https:
36 manifest_urls = [url.replace('https://', 'http://') for url in manifest_urls]
37
38 manifest_bytes = None
39 for url in manifest_urls:
40 self.log.debug(f'Trying to download manifest from {url} ...')
41 r = self.egs.unauth_session.get(url)
42 if r.ok:
43 manifest_bytes = r.content
44 break
45 else:
46 self.log.warning(f'Unable to download manifest from {url}, trying next one ...')
47 if not manifest_bytes:
48 raise ValueError('Unable to get manifest data from any CDN URL')
49
50 if sha1(manifest_bytes).hexdigest() != manifest_hash:
51 raise ValueError('Manifest sha hash mismatch!')
52
53 return manifest_bytes, base_urls
