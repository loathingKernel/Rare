61def load_color_scheme(path: str) -> QPalette:
62 palette = QPalette()
63 scheme = QSettings(path, QSettings.IniFormat)
64 try:
65 scheme.beginGroup("ColorScheme")
66 for g in color_group_map:
67 scheme.beginGroup(color_group_map[g])
68 group = QPalette.ColorGroup(g)
69 for r in color_role_map:
70 role = QPalette.ColorRole(r)
71 color = scheme.value(color_role_map[r], None)
72 if color is not None:
73 palette.setColor(group, role, QColor(color))
74 else:
75 palette.setColor(group, role, palette.color(QPalette.Active, role))
76 scheme.endGroup()
77 scheme.endGroup()
78 except:
79 palette = None
80 return palette
