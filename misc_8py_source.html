60def load_color_scheme(path: str) -> QPalette:
61 palette = QPalette()
62 scheme = QSettings(path, QSettings.IniFormat)
63 try:
64 scheme.beginGroup("ColorScheme")
65 for g in color_group_map:
66 scheme.beginGroup(color_group_map[g])
67 group = QPalette.ColorGroup(g)
68 for r in color_role_map:
69 role = QPalette.ColorRole(r)
70 color = scheme.value(color_role_map[r], None)
71 if color is not None:
72 palette.setColor(group, role, QColor(color))
73 else:
74 palette.setColor(group, role, palette.color(QPalette.Active, role))
75 scheme.endGroup()
76 scheme.endGroup()
77 except:
78 palette = None
79 return palette
