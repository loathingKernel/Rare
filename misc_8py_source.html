59def load_color_scheme(path: str) -> QPalette:
60 palette = QPalette()
61 scheme = QSettings(path, QSettings.IniFormat)
62 try:
63 scheme.beginGroup("ColorScheme")
64 for g in color_group_map:
65 scheme.beginGroup(color_group_map[g])
66 group = QPalette.ColorGroup(g)
67 for r in color_role_map:
68 role = QPalette.ColorRole(r)
69 color = scheme.value(color_role_map[r], None)
70 if color is not None:
71 palette.setColor(group, role, QColor(color))
72 else:
73 palette.setColor(group, role, palette.color(QPalette.Active, role))
74 scheme.endGroup()
75 scheme.endGroup()
76 except:
77 palette = None
78 return palette

179 def __init__(self, core: LegendaryCore):
180 super(CloudWorker, self).__init__()

185 def run(self) -> None:
186 try:
187 saves = self.corecore.get_save_games()
188 except HTTPError:
189 self.signalssignals.result_ready.emit(None)
190 return
191
192 save_games = set()
193 for igame in self.corecore.get_installed_list():
194 game = self.corecore.get_game(igame.app_name)
195 if self.corecore.is_installed(igame.app_name) and game.supports_cloud_saves:
196 save_games.add(igame.app_name)
197
198 latest_saves = dict()
199 for s in sorted(saves, key=lambda a: a.datetime):
200 if s.app_name in save_games:
201 if not latest_saves.get(s.app_name):
202 latest_saves[s.app_name] = []
203 latest_saves[s.app_name].append(s)
204
205 self.signalssignals.result_ready.emit(latest_saves)
