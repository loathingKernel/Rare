156def apply_workarounds(app_name: str):
157 if workaround := __workarounds.get(app_name):
158 # apply options
159 for opt in (options := workaround.get("options", {})):
160 if config.get_option(app_name, opt, None) is not None:
161 continue
162 if platform.system() not in options[opt].get("os", set()):
163 continue
164 config.set_option(app_name, opt, __subst(options[opt]["value"]))
165 # apply environment
166 for var in (environ := workaround.get("environ", {})):
167 if config.get_envvar(app_name, var, None) is not None:
168 continue
169 if platform.system() not in environ[var].get("os", set()):
170 continue
171 config.set_envvar(app_name, var, __subst(environ[var]["value"]))
