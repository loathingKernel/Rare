337class InstallInfoWorker(QRunnable):
338 class Signals(QObject):
339 result = pyqtSignal(InstallDownloadModel)
340 failed = pyqtSignal(str)
341 finished = pyqtSignal()
342
343 def __init__(self, core: LegendaryCore, dl_item: InstallQueueItemModel, game: Game = None):
344 sys.excepthook = sys.__excepthook__
345 super(InstallInfoWorker, self).__init__()
346 self.signalssignals = InstallInfoWorker.Signals()
347 self.corecore = core
348 self.dl_itemdl_item = dl_item
349 self.gamegame = game

351 @pyqtSlot()
352 def run(self):
353 try:
354 if not self.dl_itemdl_item.options.overlay:
355 cli = LegendaryCLI(self.corecore)
356 status = LgndrIndirectStatus()
357 result = cli.install_game(
358 LgndrInstallGameArgs(**self.dl_itemdl_item.options.as_install_kwargs(), indirect_status=status)
359 )
360 if result:
361 download = InstallDownloadModel(*result)
362 else:
363 raise LgndrException(status.message)
364 else:
365 if not os.path.exists(path := self.dl_itemdl_item.options.base_path):
366 os.makedirs(path)
367
368 dlm, analysis, igame = self.corecore.prepare_overlay_install(
369 path=self.dl_itemdl_item.options.base_path
370 )
371
372 download = InstallDownloadModel(
373 dlm=dlm,
374 analysis=analysis,
375 igame=igame,
376 game=self.gamegame,
377 repair=False,
378 repair_file="",
379 res=ConditionCheckResult(), # empty
380 )
381
382 if not download.res or not download.res.failures:
383 self.signalssignals.result.emit(download)
384 else:
385 self.signalssignals.failed.emit("\n".join(str(i) for i in download.res.failures))
386 except LgndrException as ret:
387 self.signalssignals.failed.emit(ret.message)
388 except Exception as e:
389 self.signalssignals.failed.emit(str(e))
390 self.signalssignals.finished.emit()
