13def blob_to_num(in_str):
14 """
15 The JSON manifest use a rather strange format for storing numbers.
16
17 It's essentially %03d for each char concatenated to a string.
18 ...instead of just putting the fucking number in the JSON...
19
20 Also it's still little endian so we have to bitshift it.
21
22 """
23 num = 0
24 shift = 0
25 for i in range(0, len(in_str), 3):
26 num += (int(in_str[i:i + 3]) << shift)
27 shift += 8
28 return num
