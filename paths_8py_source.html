122def get_rare_executable() -> List[str]:
123 # lk: detect if nuitka
124 if "__compiled__" in globals():
125 executable = [sys.executable]
126 elif platform.system() == "Linux" or platform.system() == "Darwin":
127 if p := os.environ.get("APPIMAGE"):
128 executable = [p]
129 else:
130 if sys.executable == os.path.abspath(sys.argv[0]):
131 executable = [sys.executable]
132 else:
133 executable = [sys.executable, os.path.abspath(sys.argv[0])]
134 elif platform.system() == "Windows":
135 executable = [sys.executable]
136
137 if sys.executable != os.path.abspath(sys.argv[0]):
138 executable.append(os.path.abspath(sys.argv[0]))
139
140 if executable[0].endswith("python.exe"):
141 # be sure to start consoleless then
142 executable[0] = executable[0].replace("python.exe", "pythonw.exe")
143 if executable[1].endswith("rare"):
144 executable[1] = executable[1] + ".exe"
145 else:
146 executable = [sys.executable]
147
148 executable[0] = os.path.abspath(executable[0])

152def create_desktop_link(app_name: str, app_title: str = "", link_name: str = "", link_type="desktop") -> bool:
153 """
154 Creates a desktop or start menu shortcut link
155
156 :param app_name:
157 app_name or "rare_shortcut" for Rare itself
158 :param app_title:
159 the title shown in the shortcut
160 (overrides to "Rare" for "rare_shortcut")
161 :param link_name:
162 the sanitized filename of the shortcut (use the folder_name attribute of RareGame)
163 (overrides to "Rare" for "rare_shortcut")
164 :param link_type:
165 "desktop" or "start_menu"
166 :return bool:
167 True if successful else False
168 """
169 # macOS is based on Darwin
170 if not desktop_links_supported():
171 logger.error(f"Shortcut creation is not available on {platform.system()}")
172 return False
173
174 if link_type not in ["desktop", "start_menu"]:
175 logger.error(f"Invalid link type {link_type}")
176 return False
177
178 for_rare = app_name == "rare_shortcut"
179
180 if for_rare:
181 icon_path = resources_path.joinpath("images", f"Rare.{desktop_icon_suffix()}")
182 app_title = "Rare"
183 link_name = "Rare"
184 else:
185 icon_path = image_dir().joinpath(app_name, f"icon.{desktop_icon_suffix()}")
186 if not app_title or not link_name:
187 logger.error("Missing app_title or link_name")
188 return False
189
190 shortcut_path = desktop_link_path(link_name, link_type)
191 if not shortcut_path.parent.exists():
192 logger.error(f"Parent directory {shortcut_path.parent} does not exist")
193 return False
194 else:
195 logger.info(f"Creating shortcut for {app_title} at {shortcut_path}")
196
197 if platform.system() == "Linux":
198 executable = get_rare_executable()
199 executable = shlex.join(executable)
200 if not for_rare:
201 executable = f"{executable} launch {app_name}"
202
203 with shortcut_path.open(mode="w") as desktop_file:
204 desktop_file.write(
205 "[Desktop Entry]\n"
206 f"Name={app_title}\n"
207 "Type=Application\n"
208 "Categories=Game;\n"
209 f"Icon={icon_path}\n"
210 f"Exec={executable}\n"
211 "Terminal=false\n"
212 "StartupWMClass=Rare\n"
213 )
214 # shortcut_path.chmod(0o755)
215 return True
216
217 elif platform.system() == "Windows":
218 # Add shortcut
219 shell = Dispatch("WScript.Shell")
220 shortcut = shell.CreateShortCut(str(shortcut_path))
221
222 executable = get_rare_executable()
223 arguments = []
224
225 if len(executable) > 1:
226 arguments.extend(executable[1:])
227 executable = executable[0]
228
229 if not for_rare:
230 arguments.extend(["launch", app_name])
231 shortcut.Targetpath = executable
232 # Maybe there is a better solution, but windows does not accept single quotes (Windows is weird)
233 shortcut.Arguments = shlex.join(arguments).replace("'", '"')
234
235 shortcut.Description = app_title
236 if for_rare:
237 shortcut.WorkingDirectory = str(home_dir())
238
239 # Icon
240 shortcut.IconLocation = icon_path.absolute()
241
242 shortcut.save()
243 return True
