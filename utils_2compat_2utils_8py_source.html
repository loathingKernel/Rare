29def prepare_process(command: List[str], environment: Dict) -> Tuple[str, List[str], Dict]:
30 logger.debug("Preparing process: %s", command)
31 _env = os.environ.copy()
32 _command = command.copy()

49def get_configured_qprocess(command: List[str], environment: Dict) -> QProcess:
