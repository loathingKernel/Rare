105 def __override_exe_edit_callback(self, path: str) -> Tuple[bool, str, int]:
106 if not path or self.igameigame is None:
107 return True, path, IndicatorReasonsCommon.VALID
108 if not os.path.isabs(path):
109 path = os.path.join(self.igameigame.install_path, path)
110 # lk: Compare paths through python's commonpath because in windows we
111 # cannot compare as strings
112 # antonia disapproves of this
113 if os.path.commonpath([self.igameigame.install_path, path]) != self.igameigame.install_path:
114 return False, self.igameigame.install_path, IndicatorReasonsCommon.WRONG_PATH
115 if not os.path.exists(path):
116 return False, path, IndicatorReasonsCommon.WRONG_PATH
117
118 if not path.endswith(".exe") and not path.endswith(".app"):
119 return False, path, IndicatorReasonsCommon.WRONG_PATH
120 path = os.path.relpath(path, self.igameigame.install_path)
121 return True, path, IndicatorReasonsCommon.VALID
