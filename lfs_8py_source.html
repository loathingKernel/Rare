32def delete_filelist(path: str, filenames: List[str],
33 delete_root_directory: bool = False,
34 silent: bool = False) -> bool:
35 dirs = set()
36 no_error = True
37
38 # delete all files that were installed
39 for filename in filenames:
40 _dir, _fn = os.path.split(filename)
41 if _dir:
42 dirs.add(_dir)
43
44 try:
45 os.remove(os.path.join(path, _dir, _fn))
46 except Exception as e:
47 if not silent:
48 logger.error(f'Failed deleting file {filename} with {e!r}')
49 no_error = False
50
51 # add intermediate directories that would have been missed otherwise
52 for _dir in sorted(dirs):
53 head, _ = os.path.split(_dir)
54 while head:
55 dirs.add(head)
56 head, _ = os.path.split(head)
57
58 # remove all directories
59 for _dir in sorted(dirs, key=len, reverse=True):
60 try:
61 os.rmdir(os.path.join(path, _dir))
62 except FileNotFoundError:
63 # directory has already been deleted, ignore that
64 continue
65 except Exception as e:
66 if not silent:
67 logger.error(f'Failed removing directory "{_dir}" with {e!r}')
68 no_error = False
69
70 if delete_root_directory:
71 try:
72 os.rmdir(path)
73 except Exception as e:
74 if not silent:
75 logger.error(f'Removing game directory failed with {e!r}')
76
77 return no_error

80def validate_files(base_path: str, filelist: List[tuple], hash_type='sha1',
81 large_file_threshold=1024 * 1024 * 512) -> Iterator[tuple]:
82 """
83 Validates the files in filelist in path against the provided hashes
84
85 :param base_path: path in which the files are located
86 :param filelist: list of tuples in format (path, hash [hex])
87 :param hash_type: (optional) type of hash, default is sha1
88 :param large_file_threshold: (optional) threshold for large files, default is 512 MiB
89 :return: yields tuples in format (VerifyResult, path, hash [hex], bytes read)
90 """
91
92 if not filelist:
93 raise ValueError('No files to validate!')
94
95 if not os.path.exists(base_path):
96 raise OSError('Path does not exist')
97
98 for file_path, file_hash in filelist:
99 full_path = os.path.join(base_path, file_path)
100 # logger.debug(f'Checking "{file_path}"...')
101
102 if not os.path.exists(full_path):
103 yield VerifyResult.FILE_MISSING, file_path, '', 0
104 continue
105
106 show_progress = False
107 interval = 0
108 speed = 0.0
109 start_time = 0.0
110
111 try:
112 _size = os.path.getsize(full_path)
113 if _size > large_file_threshold:
114 # enable progress indicator and go to new line
115 stdout.write('\n')
116 show_progress = True
117 interval = (_size / (1024 * 1024)) // 100
118 start_time = time()
119
120 with open(full_path, 'rb') as f:
121 real_file_hash = hashlib.new(hash_type)
122 i = 0
123 while chunk := f.read(1024*1024):
124 real_file_hash.update(chunk)
125 if show_progress and i % interval == 0:
126 pos = f.tell()
127 perc = (pos / _size) * 100
128 speed = pos / 1024 / 1024 / (time() - start_time)
129 stdout.write(f'\r=> Verifying large file "{file_path}": {perc:.0f}% '
130 f'({pos / 1024 / 1024:.1f}/{_size / 1024 / 1024:.1f} MiB) '
131 f'[{speed:.1f} MiB/s]\t')
132 stdout.flush()
133 i += 1
134
135 if show_progress:
136 stdout.write(f'\r=> Verifying large file "{file_path}": 100% '
137 f'({_size / 1024 / 1024:.1f}/{_size / 1024 / 1024:.1f} MiB) '
138 f'[{speed:.1f} MiB/s]\t\n')
139
140 result_hash = real_file_hash.hexdigest()
141 if file_hash != result_hash:
142 yield VerifyResult.HASH_MISMATCH, file_path, result_hash, f.tell()
143 else:
144 yield VerifyResult.HASH_MATCH, file_path, result_hash, f.tell()
145 except Exception as e:
146 logger.fatal(f'Could not verify "{file_path}"; opening failed with: {e!r}')
147 yield VerifyResult.OTHER_ERROR, file_path, '', 0
