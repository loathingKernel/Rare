411 def generate_aliases(self):
412 self.loglog.debug('Generating list of aliases...')
413
414 self.aliasesaliases = dict()
415 aliases = set()
416 collisions = set()
417 alias_map = defaultdict(set)
418
419 for app_name in self._game_metadata_game_metadata.keys():
420 game = self.get_game_metaget_game_meta(app_name)
421 if game.is_dlc:
422 continue
423 game_folder = game.metadata.get('customAttributes', {}).get('FolderName', {}).get('value', None)
424 _aliases = generate_aliases(game.app_title, game_folder=game_folder, app_name=game.app_name)
425 for alias in _aliases:
426 if alias not in aliases:
427 aliases.add(alias)
428 alias_map[game.app_name].add(alias)
429 else:
430 collisions.add(alias)
431
432 # remove colliding aliases from map and add aliases to lookup table
433 for app_name, aliases in alias_map.items():
434 alias_map[app_name] -= collisions
435 for alias in alias_map[app_name]:
436 self.aliasesaliases[alias] = app_name

445 indent=2, sort_keys=True, default=serialise_sets)
