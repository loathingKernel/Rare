376 def verify_game(self, args: Union[LgndrVerifyGameArgs, LgndrInstallGameArgs], print_command=True, repair_mode=False, repair_online=False) -> Optional[Tuple[int, int]]:
377 # Override logger for the local context to use message as part of the indirect return value
378 logger = LgndrIndirectLogger(args.indirect_status, self.loggerlogger)
379
380 args.app_name = self._resolve_aliases(args.app_name)
381 if not self.corecore.is_installed(args.app_name):
382 logger.error(f'Game "{args.app_name}" is not installed')
383 return
384
385 logger.info(f'Loading installed manifest for "{args.app_name}"')
386 igame = self.corecore.get_installed_game(args.app_name)
387 if not os.path.exists(igame.install_path):
388 logger.error(f'Install path "{igame.install_path}" does not exist, make sure all necessary mounts '
389 f'are available. If you previously deleted the game folder without uninstalling, run '
390 f'"legendary uninstall -y {igame.app_name}" and reinstall from scratch.')
391 return
392
393 manifest_data, _ = self.corecore.get_installed_manifest(args.app_name)
394 if manifest_data is None:
395 if repair_mode:
396 if not repair_online:
397 logger.critical('No manifest could be loaded, the manifest file may be missing!')
398 raise ValueError('Local manifest is missing')
399
400 logger.warning('No manifest could be loaded, the file may be missing. Downloading the latest manifest.')
401 game = self.corecore.get_game(args.app_name, platform=igame.platform)
402 manifest_data, _ = self.corecore.get_cdn_manifest(game, igame.platform)
403 # Rare: Save the manifest if we downloaded it because it was missing
404 self.corecore.lgd.save_manifest(game.app_name, manifest_data,
405 version=self.corecore.load_manifest(manifest_data).meta.build_version,
406 platform=igame.platform)
407 else:
408 logger.critical(f'Manifest appears to be missing! To repair, run "legendary repair '
409 f'{args.app_name} --repair-and-update", this will however redownload all files '
410 f'that do not match the latest manifest in their entirety.')
411 return
412
413 manifest = self.corecore.load_manifest(manifest_data)
414
415 files = sorted(manifest.file_manifest_list.elements,
416 key=lambda a: a.filename.lower())
417
418 # build list of hashes
419 if config_tags := self.corecore.lgd.config.get(args.app_name, 'install_tags', fallback=None):
420 install_tags = set(i.strip() for i in config_tags.split(','))
421 file_list = [
422 (f.filename, f.sha_hash.hex())
423 for f in files
424 if any(it in install_tags for it in f.install_tags) or not f.install_tags
425 ]
426 else:
427 file_list = [(f.filename, f.sha_hash.hex()) for f in files]
428
429 total = len(file_list)
430 total_size = sum(manifest.file_manifest_list.get_file_by_path(fm[0]).file_size
431 for fm in file_list)
432 num = processed = last_processed = 0
433 speed = 0.0
434 percentage = 0.0
435 failed = []
436 missing = []
437
438 last_update = time.time()
439
440 logger.info(f'Verifying "{igame.title}" version "{manifest.meta.build_version}"')
441 repair_file = []
442 for result, path, result_hash, bytes_read in validate_files(igame.install_path, file_list):
443 processed += bytes_read
444 percentage = (processed / total_size) * 100.0
445 num += 1
446
447 if (delta := ((current_time := time.time()) - last_update)) > 1:
448 last_update = current_time
449 speed = (processed - last_processed) / 1024 / 1024 / delta
450 last_processed = processed
451
452 if args.verify_stdout:
453 args.verify_stdout(num, total, percentage, speed)
454
455 if result == VerifyResult.HASH_MATCH:
456 repair_file.append(f'{result_hash}:{path}')
457 continue
458 elif result == VerifyResult.HASH_MISMATCH:
459 logger.error(f'File does not match hash: "{path}"')
460 repair_file.append(f'{result_hash}:{path}')
461 failed.append(path)
462 elif result == VerifyResult.FILE_MISSING:
463 logger.error(f'File is missing: "{path}"')
464 missing.append(path)
465 else:
466 logger.error(f'Other failure (see log), treating file as missing: "{path}"')
467 missing.append(path)
468
469 if args.verify_stdout:
470 args.verify_stdout(num, total, percentage, speed)
471
472 # always write repair file, even if all match
473 if repair_file:
474 repair_filename = os.path.join(self.corecore.lgd.get_tmp_path(), f'{args.app_name}.repair')
475 with open(repair_filename, 'w', encoding='utf-8') as f:
476 f.write('\n'.join(repair_file))
477 logger.debug(f'Written repair file to "{repair_filename}"')
478
479 if not missing and not failed:
480 logger.info('Verification finished successfully.')
481 return 0, 0
482 else:
483 logger.error(f'Verification failed, {len(failed)} file(s) corrupted, {len(missing)} file(s) are missing.')
484 if print_command:
485 logger.info(f'Run "legendary repair {args.app_name}" to repair your game installation.')
486 return len(failed), len(missing)
