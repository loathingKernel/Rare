89 def start_session(self, refresh_token: str = None, exchange_token: str = None,

91 if refresh_token:

102 else:
103 raise ValueError('At least one token type must be specified!')
104
105 r = self.sessionsession.post(f'https://{self._oauth_host}/account/api/oauth/token',
106 data=params, auth=self._oauth_basic_oauth_basic,
107 timeout=self.request_timeoutrequest_timeout)
108 # Only raise HTTP exceptions on server errors
109 if r.status_code >= 500:
110 r.raise_for_status()
111
112 j = r.json()
113 if 'error' in j:
114 self.loglog.warning(f'Login to EGS API failed with errorCode: {j["errorCode"]}')
115 raise InvalidCredentialsError(j['errorCode'])
116
117 self.sessionsession.headers['Authorization'] = f'bearer {j["access_token"]}'
118 # only set user info when using non-anonymous login
119 if not client_credentials:
120 self.useruser = j
