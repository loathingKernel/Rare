281 def load_pixmaps(self) -> None:
282 """
283 Load pixmaps for all games
284
285 This exists here solely to fight signal and possibly threading issues.
286 The initial image loading at startup should not be done in the RareGame class
287 for two reasons. It will delay startup due to widget updates and the image
288 might become availabe before the UI is brought up. In case of the second, we
289 will get both a long queue of signals to be serviced and some of them might
290 be not connected yet and the widget won't be updated. So do the loading here
291 by calling this after the MainWindow has finished initializing.
292
293 @return: None
294 """
295 self.threadpoolthreadpool.start(self.__load_pixmaps__load_pixmaps)
