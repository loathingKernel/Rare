24def find_app_name(path: str, core) -> Optional[str]:
25 if os.path.exists(os.path.join(path, ".egstore")):
26 for i in os.listdir(os.path.join(path, ".egstore")):
27 if i.endswith(".mancpn"):
28 with open(os.path.join(path, ".egstore", i)) as file:
29 app_name = json.load(file).get("AppName")
30 return app_name
31 elif app_name := LegendaryCLI(core).resolve_aliases(os.path.basename(os.path.normpath(path))):
32 # return None if game does not exist (Workaround for overlay)
33 if not core.get_game(app_name):
34 return None
35 return app_name
36 else:
37 logger.warning(f"Could not find AppName for {path}")
38 return None

71 def run(self) -> None:
72 result_list: List = []
73 if self.import_folderimport_folder:
74 folders = [i for i in self.pathpath.iterdir() if i.is_dir()]
75 number_of_folders = len(folders)
76 for i, child in enumerate(folders):
77 if not child.is_dir():
78 continue
79 result = self.__try_import__try_import(child, None)
80 result_list.append(result)
81 self.signalssignals.progress.emit(int(100 * i // number_of_folders))
82 else:
83 result = self.__try_import__try_import(self.pathpath, self.app_nameapp_name)
84 result_list.append(result)

87 def __try_import(self, path: Path, app_name: str = None) -> ImportedGame:
88 result = ImportedGame(ImportResult.ERROR)
89 result.path = str(path)
90 if app_name or (app_name := find_app_name(str(path), self.corecore)):
91 result.app_name = app_name
92 result.app_title = app_title = self.corecore.get_game(app_name).app_title
93 success, message = self.__import_game__import_game(path, app_name, app_title)
94 if not success:
95 result.result = ImportResult.FAILED
96 result.message = message
97 else:
98 result.result = ImportResult.SUCCESS
99 return result
