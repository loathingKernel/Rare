67 """
68 Finds an optimal background color for text on the image by calculating the
69 average color of the image and inverting it.
70
71 The image list is supposed to be a one-dimensional list of arbitrary length
72 containing RGB tuples, ranging from 0 to 255.
73 """
74 # cursed, I know
75 average = map(lambda value: value // len(pixel_list), map(sum, zip(*pixel_list)))
76 inverted = map(lambda value: 255 - value, average)

89 def text_color_for_background(background: Tuple[int, int, int]) -> Tuple[int, int, int]:
90 """
91 Calculates whether a black or white text color would fit better for the
92 given background, and returns that color. This is done by calculating the
93 luminance and simple comparing of bounds
94 """
95 # see https://alienryderflex.com/hsp.html
96 (red, green, blue) = background
97 luminance = math.sqrt(0.299 * red**2 + 0.587 * green**2 + 0.114 * blue**2)
98 if luminance < 127:
99 return 255, 255, 255
100 else:
101 return 0, 0, 0
