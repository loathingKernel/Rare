170 def doLayout(self, rect, testonly):
171 """!
172 @brief Arranges the widgets for this layout
173
174 <pre>
175 Layout
176 +-----------------------------------------------+
177 | margin (above first row only) |
178 |-----------------------------------------------|
179 | vspace (hspace) |
180 |-----------------------------------------------|
181 | hpadding |
182 |-----------------------------------------------|
183 | _ _ __ +--------+ _ __ +--------+ _ __ _ |
184 ||m||h|| h|| Widget ||h|| h|| Widget ||h|| h||m||
185 ||a||s|| p|| ||s|| p|| ||s|| p||a||
186 ||r||p|| a|| ||p|| a|| ||p|| a||r||
187 ||g||a|| d|| ||a|| d|| ||a|| d||g||
188 ||i||c|| d|| ||c|| d|| ||c|| d||i||
189 ||n||e|| i|| ||e|| i|| ||e|| i||n||
190 || || || n|| || || n|| || || n|| ||
191 || || || g|| || || g|| || || g|| ||
192 | - - -- +--------+ - -- +--------+ - -- - |
193 |-----------------------------------------------|
194 | vspace (hspace) |
195 |-----------------------------------------------|
196 | hpadding |
197 |-----------------------------------------------|
198 | margin (below last row only) |
199 +-----------------------------------------------+
200
201 margin: doesn't play a role in the code below, it only affects the
202 effective rectangle inside which we can layout
203 hspace: static padding between widgets (minimum distance between them)
204 hpadding: dynamic padding to fill the space when resizing
205 </pre>
206
207 @param self: object self-reference
208 @param rect: the area the widgets should occupy
209 @param testonly: only test the layout, don't arrange the items
210
211 @return: the height of the layout
212 """
213
214 left, top, right, bottom = self.getContentsMargins()
215 effective = rect.adjusted(+left, +top, -right, -bottom)
216 x = effective.x()
217 y = effective.y()
218 lineheight = 0
219
220 if not self._items_items:
221 return y + lineheight - rect.y() + bottom
222
223 widget = self._items_items[0].widget()
224 hspace = self.horizontalSpacinghorizontalSpacing()
225 if hspace == -1:
226 hspace = widget.style().layoutSpacing(
227 QSizePolicy.PushButton, QSizePolicy.PushButton, Qt.Horizontal
228 )
229 vspace = self.verticalSpacingverticalSpacing()
230 if vspace == -1:
231 vspace = widget.style().layoutSpacing(
232 QSizePolicy.PushButton, QSizePolicy.PushButton, Qt.Vertical
233 )
234
235 # lk: get the remaining space after subtracting the space required for each widget and its static padding
236 # lk: also reserve space for the leading static padding
237 rem_hspace = (effective.width() - hspace) % (widget.size().width() + hspace)
238 # lk: the number of items and their static spacing that can be in the layout
239 hspace_items = (effective.width() - hspace) // (widget.size().width() + hspace)
240
241 # lk: in case the visible items are less than the maximum possible widgets
242 visible_items = len([item for item in self._items_items if not item.isEmpty()])
243 if visible_items < hspace_items:
244 hspace_items = visible_items
245 rem_hspace = (effective.width() - hspace) - ((widget.size().width() + hspace) * hspace_items)
246
247 try:
248 # lk: the dynamic padding between each item, also account for the leading dynamic padding
249 hpadding = rem_hspace // (hspace_items + 1)
250 except ZeroDivisionError:
251 hpadding = 0
252
253 for item in self._items_items:
254 if item.isEmpty():
255 continue
256 # lk: compute the location of the next widget
257 next_x = x + item.sizeHint().width() + hspace + hpadding
258 # lk: find out if there is enough space for the widget in this row
259 # lk: account for the leading static and dynamic padding too (at the start)
260 if next_x - hspace * 2 - hpadding * 2 > effective.right() and lineheight > 0:
261 x = effective.x()
262 # lk: find next vertical position, add static and dynamic padding
263 y = y + lineheight + vspace + hpadding
264 next_x = x + item.sizeHint().width() + hspace + hpadding
265 lineheight = 0
266 # lk: add static and dynamic padding to the current widget
267 x = x + hspace + hpadding
268 if not testonly:
269 item.setGeometry(QRect(QPoint(x, y), item.sizeHint()))
270 x = next_x
271 lineheight = max(lineheight, item.sizeHint().height())
272 return y + lineheight - rect.y() + bottom
