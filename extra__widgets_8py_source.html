99 def minimumSize(self):
100 size = QSize()
101 for item in self._items_items:
102 size = size.expandedTo(item.minimumSize())
103 left, top, right, bottom = self.getContentsMargins()
104 size += QSize(left + right, top + bottom)
105 return size

107 def doLayout(self, rect, testonly):
108 left, top, right, bottom = self.getContentsMargins()
109 effective = rect.adjusted(+left, +top, -right, -bottom)
110 x = effective.x()
111 y = effective.y()
112 lineheight = 0
113 for item in self._items_items:
114 widget = item.widget()
115 if not widget.isVisible():
116 continue
117 hspace = self.horizontalSpacinghorizontalSpacing()
118 if hspace == -1:
119 hspace = widget.style().layoutSpacing(
120 QSizePolicy.PushButton, QSizePolicy.PushButton, Qt.Horizontal
121 )
122 vspace = self.verticalSpacingverticalSpacing()
123 if vspace == -1:
124 vspace = widget.style().layoutSpacing(
125 QSizePolicy.PushButton, QSizePolicy.PushButton, Qt.Vertical
126 )
127 nextX = x + item.sizeHint().width() + hspace
128 if nextX - hspace > effective.right() and lineheight > 0:
129 x = effective.x()
130 y = y + lineheight + vspace
131 nextX = x + item.sizeHint().width() + hspace
132 lineheight = 0
133 if not testonly:
134 item.setGeometry(QRect(QPoint(x, y), item.sizeHint()))
135 x = nextX
136 lineheight = max(lineheight, item.sizeHint().height())
137 return y + lineheight - rect.y() + bottom

140 parent = self.parent()
141 if parent is None:
142 return -1
143 elif parent.isWidgetType():

146 return parent.spacing()
