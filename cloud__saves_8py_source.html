105 def compute_save_path(self):
106 if self.rgamergame.is_installed and self.rgamergame.game.supports_cloud_saves:
107 try:
108 new_path = self.corecore.get_save_path(self.rgamergame.app_name)
109 if platform.system() != "Windows" and not os.path.exists(new_path):
110 raise ValueError(f'Path "{new_path}" does not exist.')
111 except Exception as e:
112 logger.warning(str(e))
113 resolver = WineResolver(self.corecore, self.rgamergame.raw_save_path, self.rgamergame.app_name)
114 if not resolver.wine_env.get("WINEPREFIX"):
115 self.cloud_save_path_editcloud_save_path_edit.setText("")
116 QMessageBox.warning(self, "Warning", "No wine prefix selected. Please set it in settings")
117 return
118 self.cloud_save_path_editcloud_save_path_edit.setText(self.tr("Loading..."))
119 self.cloud_save_path_editcloud_save_path_edit.setDisabled(True)
120 self.compute_save_path_buttoncompute_save_path_button.setDisabled(True)
121
122 app_name = self.rgamergame.app_name
123 resolver.signals.result_ready.connect(lambda x: self.wine_resolver_finishedwine_resolver_finished(x, app_name))
124 QThreadPool.globalInstance().start(resolver)
125 return
126 else:
127 self.cloud_save_path_editcloud_save_path_edit.setText(new_path)

143 "Error while calculating path for <b>{}</b>. Insufficient permisions to create <b>{}</b>"

146 return
