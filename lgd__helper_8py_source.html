153def prepare_process(command: List[str], environment: Dict) -> Tuple[str, List[str], Dict]:
154 logger.debug("Preparing process: %s", command)
155
156 environ = environment.copy()
157 # Sanity check environment (mostly for Linux)
158 command_line = shlex.join(command)
159 if os.environ.get("XDG_CURRENT_DESKTOP", None) == "gamescope" or "gamescope" in command_line:
160 # disable mangohud in gamescope
161 environ["MANGOHUD"] = "0"
162 # ensure shader compat dirs exist
163 if platform.system() in {"Linux", "FreeBSD"}:
164 if "WINEPREFIX" in environ and not os.path.isdir(environ["WINEPREFIX"]):
165 os.makedirs(environ["WINEPREFIX"], exist_ok=True)
166 if "STEAM_COMPAT_DATA_PATH" in environ:
167 compat_pfx = os.path.join(environ["STEAM_COMPAT_DATA_PATH"], "pfx")
168 if not os.path.isdir(compat_pfx):
169 os.makedirs(compat_pfx, exist_ok=True)
170 if "STEAM_COMPAT_SHADER_PATH" in environ:
171 environ.update(setup_compat_shaders_dir(environ["STEAM_COMPAT_SHADER_PATH"]))
172 environ["WINEDLLOVERRIDES"] = environ.get("WINEDLLOVERRIDES", "") + ";lsteamclient=d;"
173
174 _env = os.environ.copy()
175 _command = command.copy()
176
177 if os.environ.get("container") == "flatpak":
178 flatpak_command = ["flatpak-spawn", "--host"]
179 flatpak_command.extend(f"--env={name}={value}" for name, value in environ.items())
180 _command = flatpak_command + command
181 else:
182 _env.update(environ)
183
184 return _command[0], _command[1:] if len(_command) > 1 else [], _env
