54def get_origin_params(core: LegendaryCore, app_name, offline: bool,
55 launch_args: LaunchArgs) -> LaunchArgs:
56 origin_uri = core.get_origin_uri(app_name, offline)
57 if platform.system() == "Windows":
58 launch_args.executable = origin_uri
59 launch_args.args = []
60 # only set it here true, because on linux it is a launch command like every other game
61 launch_args.is_origin_game = True
62 return launch_args
63
64 command = core.get_app_launch_command(app_name)
65 if not os.path.exists(command[0]) and shutil.which(command[0]) is None:
66 return launch_args
67 command.append(origin_uri)
68
69 env = core.get_app_environment(app_name)
70 launch_args.env = QProcessEnvironment.systemEnvironment()
71 for name, value in env.items():
72 launch_args.env.insert(name, value)
73
74 launch_args.executable = command[0]
75 launch_args.args = command[1:]
76 return launch_args
