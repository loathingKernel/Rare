135def unpad(plaintext):
136 """
137 Removes a PKCS#7 padding, returning the unpadded text and ensuring the
138 padding was correct.
139 """
140 padding_len = plaintext[-1]
141 assert padding_len > 0
142 message, padding = plaintext[:-padding_len], plaintext[-padding_len:]
143 assert all(p == padding_len for p in padding)

152class AES:
153 """
154 Class for AES-128 encryption with CBC mode and PKCS#7.
155
156 This is a raw implementation of AES, without key stretching or IV
157 management. Unless you need that, please use `encrypt` and `decrypt`.
158 """
159 rounds_by_key_size = {16: 10, 24: 12, 32: 14}
160
161 def __init__(self, master_key):
162 """
163 Initializes the object with a given key.
164 """
165 assert len(master_key) in AES.rounds_by_key_size
166 self.n_roundsn_rounds = AES.rounds_by_key_size[len(master_key)]
167 self._key_matrices_key_matrices = self._expand_key_expand_key(master_key)
168
169 def _expand_key(self, master_key):
170 """
171 Expands and returns a list of key matrices for the given master_key.
172 """
173 # Initialize round keys with raw key material.
174 key_columns = bytes2matrix(master_key)
175 iteration_size = len(master_key) // 4
176
177 # Each iteration has exactly as many columns as the key material.
178 i = 1
179 while len(key_columns) < (self.n_roundsn_rounds + 1) * 4:
180 # Copy previous word.
181 word = list(key_columns[-1])
182
183 # Perform schedule_core once every "row".
184 if len(key_columns) % iteration_size == 0:
185 # Circular shift.
186 word.append(word.pop(0))
187 # Map to S-BOX.
188 word = [s_box[b] for b in word]
189 # XOR with first byte of R-CON, since the others bytes of R-CON are 0.
190 word[0] ^= r_con[i]
191 i += 1
192 elif len(master_key) == 32 and len(key_columns) % iteration_size == 4:
193 # Run word through S-box in the fourth iteration when using a
194 # 256-bit key.
195 word = [s_box[b] for b in word]
196
197 # XOR with equivalent word from previous iteration.
198 word = xor_bytes(word, key_columns[-iteration_size])
199 key_columns.append(word)
200
201 # Group key words in 4x4 byte matrices.
202 return [key_columns[4 * i: 4 * (i + 1)] for i in range(len(key_columns) // 4)]
203
204 def decrypt_block(self, ciphertext):
205 """
206 Decrypts a single block of 16 byte long ciphertext.
207 """
208 assert len(ciphertext) == 16
209
210 cipher_state = bytes2matrix(ciphertext)
211
212 add_round_key(cipher_state, self._key_matrices_key_matrices[-1])
213 inv_shift_rows(cipher_state)
214 inv_sub_bytes(cipher_state)
215
216 for i in range(self.n_roundsn_rounds - 1, 0, -1):
217 add_round_key(cipher_state, self._key_matrices_key_matrices[i])
218 inv_mix_columns(cipher_state)
219 inv_shift_rows(cipher_state)
220 inv_sub_bytes(cipher_state)
221
222 add_round_key(cipher_state, self._key_matrices_key_matrices[0])
223
224 return matrix2bytes(cipher_state)

226 def decrypt_ecb(self, ciphertext):
227 """
228 Decrypts `ciphertext` using ECB mode
229 """
230
231 blocks = []
232 for ciphertext_block in split_blocks(ciphertext, require_padding=False):
233 # CTR mode decrypt: ciphertext XOR encrypt(nonce)
234 block = self.decrypt_blockdecrypt_block(ciphertext_block)
235 blocks.append(block)
236
237 return b''.join(blocks)
