78def load_color_scheme(path: str) -> QPalette:
79 palette = QPalette()
80 scheme = QSettings(path, QSettings.IniFormat)
81 try:
82 scheme.beginGroup("ColorScheme")
83 for g in color_group_map:
84 scheme.beginGroup(color_group_map[g])
85 group = QPalette.ColorGroup(g)
86 for r in color_role_map:
87 role = QPalette.ColorRole(r)
88 color = scheme.value(color_role_map[r], None)
89 if color is not None:
90 palette.setColor(group, role, QColor(color))
91 else:
92 palette.setColor(group, role, palette.color(QPalette.Active, role))
93 scheme.endGroup()
94 scheme.endGroup()
95 except:
96 palette = None
97 return palette
