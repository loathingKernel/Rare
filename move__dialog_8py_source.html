113 def helper_func(reason: int) -> Tuple[bool, str, int]:
114 self.accept_buttonaccept_button.setEnabled(False)
115 return False, path, reason
116
117 if not self.rgamergame.install_path or not path:
118 return helper_func(MovePathEditReasons.DST_MISSING)
119
120 src_path = os.path.realpath(self.rgamergame.install_path)
121 dst_path = os.path.realpath(path)
122 dst_install_path = os.path.realpath(os.path.join(dst_path, os.path.basename(src_path)))
123
124 if not os.path.isdir(dst_path):
125 return helper_func(IndicatorReasonsCommon.DIR_NOT_EXISTS)
126
127 # Get free space on drive and size of game folder
128 _, _, free_space = shutil.disk_usage(dst_path)
129 source_size = path_size(src_path)
130
131 # Calculate from bytes to gigabytes
132 self.req_spacereq_space.setText(format_size(source_size))
133 self.avail_spaceavail_space.setText(format_size(free_space))
134
135 if not os.access(path, os.W_OK) or not os.access(self.rgamergame.install_path, os.W_OK):
136 return helper_func(MovePathEditReasons.NO_WRITE_PERM)
137
138 if src_path in {dst_path, dst_install_path}:
139 return helper_func(MovePathEditReasons.SAME_DIR)
140
141 if str(src_path) in str(dst_path):
142 return helper_func(MovePathEditReasons.DST_IN_SRC)
143
144 if str(dst_install_path) in str(src_path):
145 return helper_func(MovePathEditReasons.DST_IN_SRC)
146
147 for rgame in self.rcorercore.installed_games:
148 if not rgame.is_non_asset and rgame.install_path in path:
149 return helper_func(MovePathEditReasons.NESTED_DIR)
150
151 is_existing_dir = is_game_dir(src_path, dst_install_path)
152
153 for item in os.listdir(dst_path):
154 if os.path.basename(src_path) in os.path.basename(item):
155 if os.path.isdir(dst_install_path):
156 if not is_existing_dir:
157 self.warn_labelwarn_label.setHidden(False)
158 elif os.path.isfile(dst_install_path):
159 self.warn_labelwarn_label.setHidden(False)
160
161 if free_space <= source_size and not is_existing_dir:
162 return helper_func(MovePathEditReasons.NO_SPACE)
