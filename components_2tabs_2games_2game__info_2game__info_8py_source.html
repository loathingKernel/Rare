434 def helper_func(reason: str) -> Tuple[bool, str, str]:
435 self.move_gamemove_game.setEnabled(False)
436 return False, dir_selected, self.tr(reason)
437
438 if not self.install_pathinstall_path or not dir_selected:
439 return helper_func("You need to provide a directory.")
440
441 install_path = Path(self.install_pathinstall_path).resolve()
442 dest_path = Path(dir_selected).resolve()
443 dest_path_with_suffix = dest_path.joinpath(install_path.stem).resolve()
444
445 if not dest_path.is_dir():
446 return helper_func("Directory doesn't exist or file selected.")
447
448 # Get free space on drive and size of game folder
449 stat = os.statvfs(dest_path)
450 free_space_dest_drive = stat.f_bavail * stat.f_frsize
451 source_size = sum(f.stat().st_size for f in install_path.glob("**/*") if f.is_file())
452
453 # Calculate from bytes to gigabytes
454 free_space_dest_drive = round(free_space_dest_drive / 1000**3, 2)
455 source_size = round(source_size / 1000**3, 2)
456 self.aval_space_labelaval_space_label.setText(self.tr("Available space on disk: {}GB".format(free_space_dest_drive)))
457 self.req_space_labelreq_space_label.setText(self.tr("Required space: {}GB").format(source_size))
458
459 if not os.access(dir_selected, os.W_OK) or not os.access(self.install_pathinstall_path, os.W_OK):
460 return helper_func("No write permission on destination path/current install path.")
461
462 if install_path == dest_path or install_path == dest_path_with_suffix:
463 return helper_func("Same directory or parent directory selected.")
464
465 if str(install_path) in str(dest_path):
466 return helper_func("You can't select a directory that is inside the current install path.")
467
468 if str(dest_path_with_suffix) in str(install_path):
469 return helper_func("You can't select a directory which contains the game installation.")
470
471 for game in self.corecore.get_installed_list():
472 if game.install_path in dir_selected:
473 return helper_func("Game installations cannot be nested due to unintended sideeffects.")
474
475 is_existing_dir = is_game_dir(install_path, dest_path_with_suffix)
476
477 for i in dest_path.iterdir():
478 if install_path.stem in i.stem:
479 if dest_path_with_suffix.is_dir():
480 if not is_existing_dir:
481 self.warn_overwritingwarn_overwriting.setHidden(False)
482 elif dest_path_with_suffix.is_file():
483 self.warn_overwritingwarn_overwriting.setHidden(False)
484
485 if free_space_dest_drive <= source_size and not is_existing_dir:
486 return helper_func("Not enough space available on drive.")
487
488 # Fallback
489 self.move_gamemove_game.setEnabled(True)
490 return True, dir_selected, str()
