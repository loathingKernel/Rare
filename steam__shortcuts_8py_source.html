29def find_steam() -> Optional[str]:
30 if platform.system() == "Windows":
31 # Find the Steam install directory or raise an error
32 try: # 32-bit
33 key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, "SOFTWARE\\Valve\\Steams")
34 except FileNotFoundError:
35 try: # 64-bit
36 key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, "SOFTWARE\\Wow6432Node\\Valve\\Steam")
37 except FileNotFoundError:
38 return None
39 return winreg.QueryValueEx(key, "InstallPath")[0]
40 # return the first valid path
41 elif platform.system() in {"Linux", "FreeBSD"}:
42 for path in steam_client_install_paths:
43 if os.path.isdir(path) and os.path.isfile(os.path.join(path, "steam.sh")):
44 return path
45 return None

48def find_steam_users(steam_path: str) -> List[SteamUser]:
49 _users = []
50 vdf_path = os.path.join(steam_path, "config", "loginusers.vdf")
51 if not os.path.exists(vdf_path):
52 return _users
53 with open(vdf_path, "r", encoding="utf-8") as f:
54 users = vdf.load(f).get("users", {})
55 for long_id, user in users.items():
56 _users.append(SteamUser(long_id, user))
