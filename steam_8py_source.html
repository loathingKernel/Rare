35# Notes:
36# Anything older than 'Proton 5.13' doesn't have the 'require_tool_appid' attribute.
37# Anything older than 'Proton 7.0' doesn't have the 'compatmanager_layer_name' attribute.
38# In addition to that, the 'Steam Linux Runtime 1.0 (scout)' runtime lists the
39# 'Steam Linux Runtime 2.0 (soldier)' runtime as a dependency and is probably what was
40# being used for any version before 5.13.
41#
42# As a result the following implementation will list versions from 7.0 onwards which honestly
43# is a good trade-off for the amount of complexity supporting everything would ensue.

76 def command(self, verb: SteamVerb = SteamVerb.DEFAULT) -> List[str]:
77 tool_path = os.path.normpath(self.tool_pathtool_path)
78 cmd = "".join([shlex.quote(tool_path), self.toolmanifest["commandline"]])
79 # NOTE: "waitforexitandrun" seems to be the verb used in by steam to execute stuff
80 # `run` is used when setting up the environment, so use that if we are setting up the prefix.
81 cmd = cmd.replace("%verb%", str(verb))
82 return shlex.split(cmd)

85 return " ".join(map(shlex.quote, self.commandcommand(verb)))

174def find_compatibility_tools(steam_path: str) -> List[CompatibilityTool]:
175 compatibilitytools_paths = {
176 "/usr/share/steam/compatibilitytools.d",
177 os.path.expanduser(os.path.join(steam_path, "compatibilitytools.d")),
178 os.path.expanduser("~/.steam/compatibilitytools.d"),
179 os.path.expanduser("~/.steam/root/compatibilitytools.d"),
180 }
181 compatibilitytools_paths = {
182 os.path.realpath(path) for path in compatibilitytools_paths if os.path.isdir(path)
183 }
184 tools = []
185 for path in compatibilitytools_paths:
186 for entry in os.scandir(path):
187 entry_path = os.path.join(path, entry.name)
188 if entry.is_dir():
189 tool_vdf = os.path.join(entry_path, "compatibilitytool.vdf")
190 elif entry.is_file() and entry.name.endswith(".vdf"):
191 tool_vdf = entry_path
192 else:
193 continue
194
195 if not os.path.isfile(tool_vdf):
196 continue
197
198 with open(tool_vdf, "r") as f:
199 compatibilitytool = vdf.load(f)
200
201 entry_tools = compatibilitytool["compatibilitytools"]["compat_tools"]
202 for entry_tool in entry_tools.values():
203 if entry_tool["from_oslist"] != "windows" and entry_tool["to_oslist"] != "linux":
204 continue
205
206 install_path = entry_tool["install_path"]
207 tool_path = os.path.abspath(os.path.join(os.path.dirname(tool_vdf), install_path))
208 manifest_vdf = os.path.join(tool_path, "toolmanifest.vdf")
209
210 if not os.path.isfile(manifest_vdf):
211 continue
212
213 with open(manifest_vdf, "r") as f:
214 manifest = vdf.load(f)
215
216 tools.append(
217 CompatibilityTool(
218 steam_path=steam_path,
219 tool_path=tool_path,
220 toolmanifest=manifest["manifest"],
221 compatibilitytool=entry_tool,
222 )
223 )
224 return tools

282) -> Dict:

307def find_tools() -> List[Union[ProtonTool, CompatibilityTool]]:
308 steam_path = find_steam()
309 logger.debug("Using Steam install in %s", steam_path)
310 steam_libraries = find_libraries(steam_path)
311 logger.debug("Searching for tools in libraries:")
312 logger.debug("%s", steam_libraries)
313
314 runtimes = {}
315 for library in steam_libraries:
316 runtimes.update(find_runtimes(steam_path, library))
317
318 tools = []
319 for library in steam_libraries:
320 tools.extend(find_protons(steam_path, library))
321 tools.extend(find_compatibility_tools(steam_path))
322
323 for tool in tools:
324 runtime = find_runtime(tool, runtimes)
325 tool.runtime = runtime
326
327 tools = list(filter(lambda t: bool(t), tools))
328
329 return tools
